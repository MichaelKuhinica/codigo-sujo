Resultado ao executar os testes
Escopo de variável
  • Criar variáveis globais é mais fácil e perigoso do que parece;
  • Váriaveis tem escopo por função e não por bloco;
cachorro = "Bidu"; // variável global
function cebolinha(){
	cachorro = "Floquinho"; // variável global
}

var porco = "Chovinista"; // variável global
function chico_bento(){
	var porco = "Torresmo"; // variável local
}
Hoisting de variável
  • As variáveis são rastreadas e "declaradas" automaticamente no topo do escopo;
  • Pode gerar erros lógicos de interpretação;
pokemon = "Pikachu"; // variável global
alert(pokemon); // "Pikachu"
function ash(){
	alert(pokemon); // "undefined"
	var pokemon = "Squirtle"; // variável local
	alert(pokemon); // "Squirtle"
}
ash();
alert(pokemon); // "Pikachu"
Como é interpretado 
pokemon = "Pikachu"; // variável global
alert(pokemon); // "Pikachu"
function ash(){
	var pokemon;
	alert(pokemon); // "undefined"
	pokemon = "Squirtle"; // variável local
	alert(pokemon); // "Squirtle"
}
ash();
alert(pokemon); // "Pikachu"
Declaração de variáveis espalhadas no escopo
  • Declarar todas as variáveis no início do escopo;
  • Evita erros causados pelo hoisting;
function looping(){
  divs = ["sidebar", "aviso", "news_1", "footer"];
  ...
  for(i = 0; i < divs.length; i++) {
    $("#"+divs[i]).removeClass("ativo");
  }
}
Padrão Single var 
function looping(){
  var divs, i, max;
  divs = ["sidebar", "aviso", "news_1", "footer"];
  ...
  for(i = 0, max = divs.length; i < max; i++) {
    $("#"+divs[i]).removeClass("ativo");
  }  
}
Código não estruturado
  • Tenha controle do escopo das variáveis;
  • Encapsule as funções por funcionalidade;
  • Código independente;
  • Variável undefined segura;
aviso_id = "#aviso";
function someAviso() {
  $(aviso_id).fadeOut('slow');
}
function mudaAviso(msg) {
  $(aviso_id).html(msg);
}
Padrão Module 
var avisoModule = (function($, undefined){
  var aviso_id = "#aviso";
  return {
    someAviso: function() { 
      $(aviso_id).fadeOut('slow');
    },
    mudaAviso: function(msg) {
      $(aviso_id).html(msg);
    }
  };
})(jQuery);

avisoModule.mudaAviso("Teste");
Código não estruturado
  • Variável global acessível em qualquer parte da aplicação;
  • Funções soltas que utilizam a variável global;
var pilha = [];
function adicionar(item) {
    pilha.push(item);
}
function retirar() {
     return pilha.pop();
}
function tamanho() {
     return pilha.length;
}
Problemas no testes
  • Dependência entre os casos de testes;
  • Dependência não explícita;
module("Pilha sem estrutura");
test("Tamanho", function() {
    adicionar("item1");
    equal(tamanho(), 1);
});

test("Retirar", function() {
    adicionar("item1");
    adicionar("item2")
    equal(retirar(), "item2");
});
Padrão Singleton 
var appLista = (function() {
    var pilha = [];
    return {
        adicionar: function (item) {
            pilha.push(item);
        },
        retirar: function() {
            return pilha.pop();
        },
        tamanho: function() {
            return pilha.length;
        }
    };
}());
  
module("Pilha com Singleton");
test("Tamanho", function() {
    appLista.adicionar("item1");
    equal(appLista.tamanho(), 1);
});

test("Retirar", function() {
    appLista.adicionar("item1");
    appLista.adicionar("item2")
    equal(appLista.retirar(), "item2");
});
Padrão "Construtor" 
function novaAppLista() {
    var pilha = [];
    return {
        adicionar: function (item) {
            pilha.push(item);
        },
        retirar: function() {
            return pilha.pop();
        },
        tamanho: function() {
            return pilha.length;
        }
    };
}

var appLista = novaAppLista();
  
module("Pilha com Contrutor");
test("Retirar", function() {
    var appLista = novaAppLista();
    appLista.adicionar("item1");
    appLista.adicionar("item2")
    equal(appLista.retirar(), "item2");
});

test("Tamanho", function() {
    var appLista = novaAppLista();
    appLista.adicionar("item1");
    equal(appLista.tamanho(), 1);
});
