Padrão Singleton 
var armazenamento = (function() {
    var pilha = [];
    return {
        adicionar: function (item) {
            pilha.push(item);
        },
        retirar: function() {
            return pilha.pop();
        },
        tamanho: function() {
            return pilha.length;
        }
    };
}());
module("Armazenamento");
test("Tamanho", function() {
    armazenamento.adicionar("item1");
    equal(armazenamento.tamanho(), 1);
});

test("Retirar", function() {
    armazenamento.adicionar("item1");
    armazenamento.adicionar("item2")
    equal(armazenamento.retirar(), "item2");
});
Padrão "Construtor" 
function novoArmazenamento() {
    var pilha = [];
    return {
        adicionar: function (item) {
            pilha.push(item);
        },
        retirar: function() {
            return pilha.pop();
        },
        tamanho: function() {
            return pilha.length;
        }
    };
}

var armazenamento = novoArmazenamento();
module("Armazenamento");
test("Retirar", function() {
    var armazenamento = novoArmazenamento();
    armazenamento.adicionar("item1");
    armazenamento.adicionar("item2")
    equal(armazenamento.retirar(), "item2");
});

test("Tamanho", function() {
    var armazenamento = novoArmazenamento();
    armazenamento.adicionar("item1");
    equal(armazenamento.tamanho(), 1);
});
